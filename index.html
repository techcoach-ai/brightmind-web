if (!response.ok) throw new Error('Proxy error'); const parsed = await response.json(); // Use returned data or fallbacks const text = parsed.message || "Yeah, that's a common challenge in Hastings!\n\nAn AI agent can automate this 24/7.\n\nYou'll see a massive difference in your weekly admin time."; } catch(e) { console.error("AI Demo Error:", e); // Fallback so the site never looks broken renderResponse( "Honestly, we hear that a lot!\n\nA custom Enquiry Agent would solve this perfectly.\n\nYou'd save hours of back-and-forth every single week.", "5 hrs", "recovered weekly", "average for East Sussex SMEs" ); } } function renderResponse(text, num, label, sub) { document.getElementById('demo-typing').style.display = 'none'; const responseEl = document.getElementById('demo-response-text'); responseEl.style.display = 'block'; } } type(); } function showStats(num, label, sub) { const statEl = document.getElementById('demo-stat'); document.getElementById('demo-stat-num').textContent = num; document.getElementById('demo-stat-label').textContent = label; document.getElementById('demo-stat-sub').textContent = sub; statEl.style.display = 'flex'; document.getElementById('demo-ctas').style.display = 'flex'; } function resetDemo() { document.getElementById('demo-step-3').style.display = 'none';
